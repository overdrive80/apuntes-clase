- Todos las secciones de  cursos de Java empiezan el 3 de septiembre del 2007.
#+begin_src sql
update section
set start_date_time=to_date('03-09-2007','DD/MM/YYYY')
where
  section_id = any(
    select s.section_id
    from 
      course c, section s
    where
     lower(c.description) like '%java%' and
     s.course_no = c.course_no);
#+end_src
